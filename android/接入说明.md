# Android 无障碍 App 接入说明

## 概述

Android App 作为执行端，通过 WebSocket 与 PC 端 AccessAgent 通信。
不需要 USB 调试，用户只需安装 App 并开启无障碍权限。

---

## 通信协议

### PC → App（指令）

```json
// 下发任务
{"type": "task", "task": "打开微信发消息给张三"}

// 请求当前界面状态
{"type": "request_state"}

// 请求截图（关键步骤才触发）
{"type": "request_screenshot"}

// 执行点击
{"type": "click", "x": 540, "y": 960}

// 执行长按
{"type": "long_click", "x": 540, "y": 960}

// 执行输入
{"type": "type", "x": 540, "y": 960, "text": "你好"}

// 执行滚动
{"type": "scroll", "x": 540, "y": 960, "direction": "up"}

// 返回键
{"type": "back"}

// 回主屏
{"type": "home"}

// 任务完成
{"type": "finish", "message": "任务完成"}
```

### App → PC（数据）

```json
// 当前界面状态
{
  "type": "state",
  "package": "com.tencent.mm",
  "screen_size": [1080, 2340],
  "ui_elements": [
```

## Android App 核心代码结构

### 1. AccessibilityService 实现

```kotlin
class AgentAccessibilityService : AccessibilityService() {

    override fun onAccessibilityEvent(event: AccessibilityEvent) {
        // 界面变化时自动触发，可以通知 WebSocket 客户端
    }

    override fun onInterrupt() {}

    // 获取当前界面所有可交互元素
    fun getUIElements(): List<UIElement> {
        val root = rootInActiveWindow ?: return emptyList()
        val elements = mutableListOf<UIElement>()
        traverseNode(root, elements, index = intArrayOf(1))
        return elements
    }

    private fun traverseNode(
        node: AccessibilityNodeInfo,
        result: MutableList<UIElement>,
        index: IntArray
    ) {
        val bounds = Rect()
        node.getBoundsInScreen(bounds)

        if (node.isClickable || node.isEditable || node.isScrollable) {
            result.add(UIElement(
                index = index[0]++,
                className = node.className?.toString() ?: "",
                text = node.text?.toString() ?: "",
                contentDesc = node.contentDescription?.toString() ?: "",
                resourceId = node.viewIdResourceName ?: "",
                bounds = listOf(bounds.left, bounds.top, bounds.right, bounds.bottom),
                clickable = node.isClickable,
                editable = node.isEditable,
                scrollable = node.isScrollable
            ))
        }

        for (i in 0 until node.childCount) {
            node.getChild(i)?.let { traverseNode(it, result, index) }
        }
    }

    // 执行点击
    fun performClick(x: Int, y: Int) {
        val path = Path().apply { moveTo(x.toFloat(), y.toFloat()) }
        val stroke = GestureDescription.StrokeDescription(path, 0, 100)
        dispatchGesture(GestureDescription.Builder().addStroke(stroke).build(), null, null)
    }

    // 执行滑动
    fun performScroll(x: Int, y: Int, direction: String) {
        val (endX, endY) = when (direction) {
            "up"    -> x to y - 500
            "down"  -> x to y + 500
            "left"  -> x - 500 to y
            "right" -> x + 500 to y
            else    -> x to y
        }
        val path = Path().apply {
            moveTo(x.toFloat(), y.toFloat())
            lineTo(endX.toFloat(), endY.toFloat())
        }
        val stroke = GestureDescription.StrokeDescription(path, 0, 300)
        dispatchGesture(GestureDescription.Builder().addStroke(stroke).build(), null, null)
    }

    // 截图
    fun takeScreenshot(callback: (Bitmap) -> Unit) {
        // Android 9+ 支持 takeScreenshot API
        // 或使用 MediaProjection（需用户授权）
    }
}
```

### 2. WebSocket 客户端

```kotlin
class WebSocketClient(private val serverUrl: String) {

    private val client = OkHttpClient()
    private var webSocket: WebSocket? = null
    lateinit var service: AgentAccessibilityService

    fun connect() {
        val request = Request.Builder().url(serverUrl).build()
        webSocket = client.newWebSocket(request, object : WebSocketListener() {

            override fun onMessage(webSocket: WebSocket, text: String) {
                val msg = JSONObject(text)
                handleCommand(msg)
            }
        })
    }

    private fun handleCommand(msg: JSONObject) {
        when (msg.getString("type")) {
            "request_state" -> sendState()
            "request_screenshot" -> sendScreenshot()
            "click" -> {
                service.performClick(msg.getInt("x"), msg.getInt("y"))
                sendResult("success")
            }
            "type" -> {
                // 点击输入框后输入文字
                service.performClick(msg.getInt("x"), msg.getInt("y"))
                Thread.sleep(300)
                insertText(msg.getString("text"))
                sendResult("success")
            }
            "scroll" -> {
                service.performScroll(
                    msg.getInt("x"), msg.getInt("y"), msg.getString("direction")
                )
                sendResult("success")
            }
            "back" -> {
                service.performGlobalAction(AccessibilityService.GLOBAL_ACTION_BACK)
                sendResult("success")
            }
            "home" -> {
                service.performGlobalAction(AccessibilityService.GLOBAL_ACTION_HOME)
                sendResult("success")
            }
        }
    }

    private fun sendState() {
        val elements = service.getUIElements()
        val json = JSONObject().apply {
            put("type", "state")
            put("package", service.rootInActiveWindow?.packageName ?: "")
            put("ui_elements", elementsToJson(elements))
        }
        webSocket?.send(json.toString())
    }

    private fun sendResult(status: String) {
        webSocket?.send(JSONObject().apply {
            put("type", "result")
            put("status", status)
        }.toString())
    }
}
```

## 用户使用步骤

```
1. 安装 AccessAgent.apk
2. 进入 设置 → 无障碍 → AccessAgent → 开启
3. 在 App 内输入 PC 端 IP 地址（同一局域网）
4. PC 端执行：python main.py --task "打开微信发消息给张三"
5. 点击 App 内"开始连接"
```

---

## 注意事项

- App 和 PC 需在**同一局域网**（或通过 ngrok 穿透）
- Android 版本需 **7.0+**（无障碍手势 API）
- 截图功能需 **Android 9+** 或用户授权 MediaProjection
- FLAG_SECURE 页面截图黑屏属正常现象，系统自动降级为纯文本模式
